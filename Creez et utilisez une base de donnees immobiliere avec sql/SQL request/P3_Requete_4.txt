SELECT SUM (mutation.valeur_fonciere) / SUM (logement.Surface_carrez) AS "Prix au m²"

-- Prix moyen du metre_carre 

--AVG ("prix au m²")

-- Aucune idée ça fonctionne pas pour le moment

FROM mutation 

INNER JOIN logement ON mutation.Logement_id = logement.id
INNER JOIN localisation ON logement.Localisation_id = localisation.id 

-- Jointure mutation <=> logement + logement <=> localisation

 WHERE code_departement in ('75','77','78','91','92','93','94','95')

-- Choix speficique des departements ile_de_france

AND logement.type_de_local = 'Maison'

-- choix du type_de_logement

;
